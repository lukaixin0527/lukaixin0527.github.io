    MySQL存储过程，记录每个表数据大小

    数据库 > MySQL 存储过程
    Created At : 2022-03-23 15:13
    Count:289 Views 👀 :
    Comment: 
    CREATE DEFINER=`root`@`%` PROCEDURE `saveTableDataSize`()
BEGIN

DECLARE merchantName VARCHAR(255); 
DECLARE datasize int; 
DECLARE done INT DEFAULT 0;
-- 数据库名称+'.'+数据库表名称,防止名称一致冲突
DECLARE cur CURSOR FOR  select concat(table_schema,'.',table_name) from information_schema.tables where TABLE_SCHEMA in('dm_analysis','dm_base','dm_business','dm_case','dm_config','dm_expressway','dm_ports','dm_transportation');
-- 游标执行完，即遍历结束。设置done的值为1
DECLARE CONTINUE HANDLER FOR NOT FOUND SET done=1; 
-- 删除今天的datasize信息
delete from `dm_analysis`.`sys_h_datasize` where REPORT_TIME = date(now());
-- 开启游标
OPEN cur;    
    -- 执行循环
  posLoop:LOOP 
        FETCH cur INTO merchantName;		
         -- 如果done的值为1，即遍历结束，结束循环    
     IF done=1 THEN LEAVE posLoop;	             
         -- 注意 if语句需要添加END IF结束IF
         END IF;
         
         set @stmt = concat('select count(1) into @dataRowTmp from ',merchantName);-- 定义语句
     PREPARE stmt from @stmt; -- 预定义sql语句,从变量中获取	
         EXECUTE stmt; -- 执行sql语句
         DEALLOCATE PREPARE stmt; -- 释放资源，后续还可以使用
         set datasize = @dataRowTmp;
         -- 开始执行
         INSERT INTO `dm_analysis`.`sys_h_datasize`(`ID`, `REPORT_TIME`, `TABLE_SCHEMA`, `TABLE_NAME`, `TABLE_COMMENT`, `TABLE_ROWS`, `DATA_LENGTH`, `GMT_CREATE_TIME`) 
         select null,date(now()),TABLE_SCHEMA, substring_index(merchantName,'.',-1), TABLE_COMMENT,  datasize ,DATA_LENGTH,now()
         from information_schema.tables T1  where TABLE_SCHEMA = substring_index(merchantName,'.',1) and table_name = substring_index(merchantName,'.',-1) ;

         -- 关闭循环	
    END LOOP posLoop; 
    -- 关闭游标
    CLOSE cur;

END
